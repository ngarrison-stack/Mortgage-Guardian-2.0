<feature>
  <name>File validation utility</name>
  <files>backend-express/utils/fileValidation.js, backend-express/__tests__/utils/fileValidation.test.js</files>
  <behavior>
    validateFileContent(buffer, claimedFileName):
    - PDF buffer (magic: %PDF) with "test.pdf" → { valid: true, detectedType: { ext: 'pdf', mime: 'application/pdf' } }
    - JPEG buffer (magic: FF D8 FF) with "photo.jpg" → { valid: true, detectedType: { ext: 'jpg', mime: 'image/jpeg' } }
    - PNG buffer (magic: 89 50 4E 47) with "scan.png" → { valid: true, detectedType: { ext: 'png', mime: 'image/png' } }
    - EXE buffer disguised as "document.pdf" → { valid: false, error: 'File content does not match claimed type' }
    - ZIP/RAR/executable buffer → { valid: false, error: 'File type not allowed' }
    - Empty buffer → { valid: false, error: 'Empty file content' }
    - Buffer exceeding size limit → { valid: false, error: 'File exceeds maximum size...' }
    - Undetectable type (plain text) with allowed extension → { valid: true, detectedType: null, warning: 'Could not verify file type from content' }

    sanitizeFileName(fileName):
    - "report.pdf" → "report.pdf" (no change)
    - "../../../etc/passwd" → "etcpasswd" (path traversal removed)
    - "file\x00name.pdf" → "filename.pdf" (null bytes removed)
    - "my document (1).pdf" → "my_document_1_.pdf" (special chars replaced)
    - "" → throws error
    - Very long name → truncated to 255 chars preserving extension

    ALLOWED_FILE_TYPES: ['pdf', 'jpg', 'jpeg', 'png', 'heic', 'tiff', 'txt']
    FILE_SIZE_LIMITS: { pdf: 20MB, jpg/jpeg/png/heic/tiff: 10MB, txt: 5MB, default: 10MB }
  </behavior>
  <implementation>
    Install file-type@^16.5.4 (last CommonJS-compatible version — v17+ is ESM-only).
    Use FileType.fromBuffer(buffer) for magic number detection. Returns { ext, mime } or undefined.
    For undetectable types (plain text), fall back to extension check against allowed list.
    Export: validateFileContent, sanitizeFileName, ALLOWED_FILE_TYPES, FILE_SIZE_LIMITS.
    Place in backend-express/utils/fileValidation.js (create utils/ directory).
  </implementation>
</feature>
